// SEO-friendly URL oluşturma
function create_slug($text) {
    $text = trim($text);
    $search = ['Ç','Ş','Ğ','Ü','İ','Ö','ç','ş','ğ','ü','ö','ı','+','#'];
    $replace = ['c','s','g','u','i','o','c','s','g','u','o','i','plus','sharp'];
    $text = str_replace($search, $replace, $text);
    $text = strtolower($text);
    $text = preg_replace('/[^a-z0-9-]/', '-', $text);
    $text = preg_replace('/-+/', '-', $text);
    $text = trim($text, '-');
    return $text;
}

// Cache fonksiyonu
function get_cache($key) {
    if (!ENABLE_CACHE) return false;
    
    try {
        $redis = new Redis();
        $redis->connect(REDIS_HOST, REDIS_PORT);
        $data = $redis->get($key);
        return $data ? json_decode($data, true) : false;
    } catch (Exception $e) {
        return false;
    }
}

function set_cache($key, $data, $ttl = REDIS_TTL) {
    if (!ENABLE_CACHE) return false;
    
    try {
        $redis = new Redis();
        $redis->connect(REDIS_HOST, REDIS_PORT);
        return $redis->setex($key, $ttl, json_encode($data));
    } catch (Exception $e) {
        return false;
    }
}

// Multi-language support
function __($key, $lang = null) {
    if (!$lang) $lang = $_SESSION['lang'] ?? DEFAULT_LANG;
    
    static $translations = [];
    
    if (empty($translations[$lang])) {
        $file = __DIR__ . "/languages/{$lang}.php";
        if (file_exists($file)) {
            $translations[$lang] = include $file;
        }
    }
    
    return $translations[$lang][$key] ?? $key;
}

// SEO Meta Tag Generator
function generate_meta_tags($title, $description, $image = null, $type = 'website') {
    $output = '';
    $output .= '<meta property="og:title" content="' . htmlspecialchars($title) . '">' . "\n";
    $output .= '<meta property="og:description" content="' . htmlspecialchars($description) . '">' . "\n";
    $output .= '<meta property="og:type" content="' . $type . '">' . "\n";
    $output .= '<meta property="og:url" content="' . current_url() . '">' . "\n";
    
    if ($image) {
        $output .= '<meta property="og:image" content="' . $image . '">' . "\n";
    }
    
    // Twitter Cards
    $output .= '<meta name="twitter:card" content="summary_large_image">' . "\n";
    $output .= '<meta name="twitter:title" content="' . htmlspecialchars($title) . '">' . "\n";
    $output .= '<meta name="twitter:description" content="' . htmlspecialchars($description) . '">' . "\n";
    
    return $output;
}

// Minify HTML
function minify_html($html) {
    if (!MINIFY_HTML) return $html;
    
    $search = [
        '/\>[^\S ]+/s',     // strip whitespaces after tags
        '/[^\S ]+\</s',     // strip whitespaces before tags
        '/(\s)+/s',         // shorten multiple whitespace sequences
        '/<!--(.|\s)*?-->/' // remove HTML comments
    ];
    
    $replace = ['>', '<', '\\1', ''];
    
    return preg_replace($search, $replace, $html);
}

// Generate breadcrumbs
function generate_breadcrumbs($items) {
    $output = '<nav aria-label="breadcrumb"><ol class="breadcrumb">';
    $output .= '<li class="breadcrumb-item"><a href="/">Ana Sayfa</a></li>';
    
    foreach ($items as $item) {
        if (isset($item['url'])) {
            $output .= '<li class="breadcrumb-item"><a href="' . $item['url'] . '">' . $item['title'] . '</a></li>';
        } else {
            $output .= '<li class="breadcrumb-item active">' . $item['title'] . '</li>';
        }
    }
    
    $output .= '</ol></nav>';
    return $output;
}

// API Call Helper
function api_call($endpoint, $params = [], $method = 'GET') {
    $ch = curl_init();
    
    if ($method == 'GET' && !empty($params)) {
        $endpoint .= '?' . http_build_query($params);
    }
    
    curl_setopt($ch, CURLOPT_URL, API_URL . $endpoint);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
    curl_setopt($ch, CURLOPT_TIMEOUT, 10);
    
    if ($method == 'POST') {
        curl_setopt($ch, CURLOPT_POST, true);
        curl_setopt($ch, CURLOPT_POSTFIELDS, json_encode($params));
        curl_setopt($ch, CURLOPT_HTTPHEADER, ['Content-Type: application/json']);
    }
    
    $response = curl_exec($ch);
    curl_close($ch);
    
    return json_decode($response, true);
}

    $data = trim($data);
    $data = stripslashes($data);
    $data = htmlspecialchars($data);
    return $data;
}

            // Update active button

            this.classList.add('active');
            
            // Show/hide sections
            if (category === 'all') {
                document.querySelectorAll('.faq-section').forEach(section => {
                    section.style.display = 'block';
                });
            } else {
                document.querySelectorAll('.faq-section').forEach(section => {
                    if (section.getAttribute('data-category') === category) {
                        section.style.display = 'block';
                    } else {

                    }
                });
            }
        });

class FlightSearchAPI {
    private $providers = [
        'amadeus' => [
            'endpoint' => 'https://api.amadeus.com/v2/shopping/flight-offers',
            'auth_endpoint' => 'https://api.amadeus.com/v1/security/oauth2/token',
            'client_id' => AMADEUS_KEY,
            'client_secret' => AMADEUS_SECRET
        ],
        // Diğer provider'lar eklenecek
    ];
    
    private $access_token = null;
    
    public function search($params) {
        // Validate parameters
        $required = ['origin', 'destination', 'departureDate', 'adults'];
        foreach ($required as $field) {
            if (!isset($params[$field])) {
                return ['error' => "Missing required field: $field"];
            }
        }
        
        // Check cache first
        $cache_key = 'flight_search_' . md5(json_encode($params));
        $cached = get_cache($cache_key);
        if ($cached) {
            return $cached;
        }
        
        // Get access token
        $this->getAccessToken();
        
        // Search flights
        $results = $this->searchAmadeus($params);
        
        // Process and normalize results
        $normalized = $this->normalizeResults($results);
        
        // Cache results
        set_cache($cache_key, $normalized, 300); // 5 minutes
        
        return $normalized;
    }
    
    private function getAccessToken() {
        $ch = curl_init($this->providers['amadeus']['auth_endpoint']);
        curl_setopt($ch, CURLOPT_POST, 1);
        curl_setopt($ch, CURLOPT_POSTFIELDS, http_build_query([
            'grant_type' => 'client_credentials',
            'client_id' => $this->providers['amadeus']['client_id'],
            'client_secret' => $this->providers['amadeus']['client_secret']
        ]));
        curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
        curl_setopt($ch, CURLOPT_HTTPHEADER, [
            'Content-Type: application/x-www-form-urlencoded'
        ]);
        
        $response = curl_exec($ch);
        curl_close($ch);
        
        $data = json_decode($response, true);
        $this->access_token = $data['access_token'] ?? null;
    }
    
    private function searchAmadeus($params) {
        $query = [
            'originLocationCode' => strtoupper($params['origin']),
            'destinationLocationCode' => strtoupper($params['destination']),
            'departureDate' => $params['departureDate'],
            'adults' => $params['adults'],
            'children' => $params['children'] ?? 0,
            'infants' => $params['infants'] ?? 0,
            'travelClass' => $params['travelClass'] ?? 'ECONOMY',
            'nonStop' => $params['nonStop'] ?? 'false',
            'currencyCode' => 'TRY',
            'max' => 50
        ];
        
        if (isset($params['returnDate'])) {
            $query['returnDate'] = $params['returnDate'];
        }
        
        $ch = curl_init($this->providers['amadeus']['endpoint'] . '?' . http_build_query($query));
        curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
        curl_setopt($ch, CURLOPT_HTTPHEADER, [
            'Authorization: Bearer ' . $this->access_token
        ]);
        
        $response = curl_exec($ch);
        $http_code = curl_getinfo($ch, CURLINFO_HTTP_CODE);
        curl_close($ch);
        
        if ($http_code !== 200) {
            return ['error' => 'API request failed'];
        }
        
        return json_decode($response, true);
    }
    
    private function normalizeResults($amadeus_data) {
        $flights = [];
        
        if (!isset($amadeus_data['data'])) {
            return ['flights' => []];
        }
        
        foreach ($amadeus_data['data'] as $offer) {
            $flight = [
                'id' => $offer['id'],
                'price' => [
                    'amount' => $offer['price']['total'],
                    'currency' => $offer['price']['currency']
                ],
                'segments' => []
            ];
            
            foreach ($offer['itineraries'] as $itinerary) {
                foreach ($itinerary['segments'] as $segment) {
                    $flight['segments'][] = [
                        'departure' => [
                            'airport' => $segment['departure']['iataCode'],
                            'time' => $segment['departure']['at']
                        ],
                        'arrival' => [
                            'airport' => $segment['arrival']['iataCode'],
                            'time' => $segment['arrival']['at']
                        ],
                        'airline' => $segment['carrierCode'],
                        'flightNumber' => $segment['number'],
                        'duration' => $segment['duration'],
                        'aircraft' => $segment['aircraft']['code'] ?? null
                    ];
                }
            }
            
            $flights[] = $flight;
        }
        
        // Sort by price
        usort($flights, function($a, $b) {
            return $a['price']['amount'] <=> $b['price']['amount'];
        });
        
        return [
            'success' => true,
            'count' => count($flights),
            'flights' => $flights
        ];
    }
}

class HotelSearchAPI {
    private $providers = [
        'expedia' => [
            'endpoint' => 'https://api.expediapartnercentral.com/v3/properties/search',
            'api_key' => EXPEDIA_API_KEY
        ]
    ];
    
    public function search($params) {
        // Validate
        $required = ['destination', 'checkin', 'checkout', 'guests'];
        foreach ($required as $field) {
            if (!isset($params[$field])) {
                return ['error' => "Missing required field: $field"];
            }
        }
        
        // Check cache
        $cache_key = 'hotel_search_' . md5(json_encode($params));
        $cached = get_cache($cache_key);
        if ($cached) {
            return $cached;
        }
        
        // Multi-provider search
        $all_results = [];
        foreach ($this->providers as $provider => $config) {
            $results = $this->searchProvider($provider, $params);
            $all_results = array_merge($all_results, $results);
        }
        
        // Normalize and deduplicate
        $normalized = $this->normalizeResults($all_results);
        
        // Apply filters
        if (isset($params['filters'])) {
            $normalized = $this->applyFilters($normalized, $params['filters']);
        }
        
        // Sort
        $normalized = $this->sortResults($normalized, $params['sort'] ?? 'price');
        
        // Cache
        set_cache($cache_key, $normalized, 300);
        
        return $normalized;
    }
    
    private function searchProvider($provider, $params) {
        switch ($provider) {
            case 'expedia':
                return $this->searchExpedia($params);
            default:
                return [];
        }
    }
    
    private function searchExpedia($params) {
        // Expedia API implementation
        $query = [
            'location' => $params['destination'],
            'check_in' => $params['checkin'],
            'check_out' => $params['checkout'],
            'occupancy' => $params['guests'],
            'currency' => 'TRY',
            'language' => 'tr-TR'
        ];
        
        // API call implementation
        // ...
        
        return []; // Placeholder
    }
    
    private function normalizeResults($results) {
        $hotels = [];
        
        foreach ($results as $hotel) {
            $hotels[] = [
                'id' => $hotel['id'],
                'name' => $hotel['name'],
                'location' => [
                    'address' => $hotel['address'],
                    'lat' => $hotel['latitude'],
                    'lng' => $hotel['longitude']
                ],
                'price' => [
                    'amount' => $hotel['price'],
                    'currency' => 'TRY'
                ],
                'rating' => $hotel['rating'] ?? 0,
                'stars' => $hotel['stars'] ?? 0,
                'amenities' => $hotel['amenities'] ?? [],
                'images' => $hotel['images'] ?? [],
                'boardType' => $hotel['board_type'] ?? 'RO'
            ];
        }
        
        return [
            'success' => true,
            'count' => count($hotels),
            'hotels' => $hotels
        ];
    }
    
    private function applyFilters($hotels, $filters) {
        // Price filter
        if (isset($filters['minPrice'])) {
            $hotels['hotels'] = array_filter($hotels['hotels'], function($hotel) use ($filters) {
                return $hotel['price']['amount'] >= $filters['minPrice'];
            });
        }
        
        if (isset($filters['maxPrice'])) {
            $hotels['hotels'] = array_filter($hotels['hotels'], function($hotel) use ($filters) {
                return $hotel['price']['amount'] <= $filters['maxPrice'];
            });
        }
        
        // Star rating filter
        if (isset($filters['stars'])) {
            $hotels['hotels'] = array_filter($hotels['hotels'], function($hotel) use ($filters) {
                return in_array($hotel['stars'], $filters['stars']);
            });
        }
        
        // Board type filter
        if (isset($filters['boardType'])) {
            $hotels['hotels'] = array_filter($hotels['hotels'], function($hotel) use ($filters) {
                return in_array($hotel['boardType'], $filters['boardType']);
            });
        }
        
        $hotels['count'] = count($hotels['hotels']);
        return $hotels;
    }
    
    private function sortResults($hotels, $sortBy) {
        switch ($sortBy) {
            case 'price':
                usort($hotels['hotels'], function($a, $b) {
                    return $a['price']['amount'] <=> $b['price']['amount'];
                });
                break;
            case 'rating':
                usort($hotels['hotels'], function($a, $b) {
                    return $b['rating'] <=> $a['rating'];
                });
                break;
            case 'stars':
                usort($hotels['hotels'], function($a, $b) {
                    return $b['stars'] <=> $a['stars'];
                });
                break;
        }
        
        return $hotels;
    }
}

    $input = json_decode(file_get_contents('php://input'), true);
    
    $api = new HotelSearchAPI();
    $results = $api->search($input);
    
    echo json_encode($results);

class PriceAlertAPI {
    private $db;
    
    public function __construct() {
        $this->db = new PDO(
            "mysql:host=" . DB_HOST . ";dbname=" . DB_NAME . ";charset=" . DB_CHARSET,
            DB_USER,
            DB_PASS
        );
    }
    
    public function create($data) {
        // Validate email
        if (!filter_var($data['email'], FILTER_VALIDATE_EMAIL)) {
            return ['error' => 'Invalid email address'];
        }
        
        // Generate unique token
        $token = bin2hex(random_bytes(32));
        
        // Insert alert
        $stmt = $this->db->prepare("
            INSERT INTO price_alerts 
            (email, type, route_from, route_to, departure_date, return_date, 
             target_price, current_price, token, created_at, status)
            VALUES 
            (:email, :type, :route_from, :route_to, :departure_date, :return_date,
             :target_price, :current_price, :token, NOW(), 'active')
        ");
        
        $stmt->execute([
            ':email' => $data['email'],
            ':type' => $data['type'], // flight, hotel, car, package
            ':route_from' => $data['from'] ?? null,
            ':route_to' => $data['to'] ?? null,
            ':departure_date' => $data['departure_date'] ?? null,
            ':return_date' => $data['return_date'] ?? null,
            ':target_price' => $data['target_price'] ?? null,
            ':current_price' => $data['current_price'] ?? null,
            ':token' => $token
        ]);
        
        // Send confirmation email
        $this->sendConfirmationEmail($data['email'], $token);
        
        return [
            'success' => true,
            'message' => 'Price alert created successfully',
            'token' => $token
        ];
    }
    
    public function checkAlerts() {
        // Get active alerts
        $stmt = $this->db->prepare("
            SELECT * FROM price_alerts 
            WHERE status = 'active' 
            AND departure_date > CURDATE()
            AND last_checked < DATE_SUB(NOW(), INTERVAL 1 HOUR)
            LIMIT 100
        ");
        
        $stmt->execute();
        $alerts = $stmt->fetchAll(PDO::FETCH_ASSOC);
        
        foreach ($alerts as $alert) {
            $this->checkAndNotify($alert);
        }
        
        return ['checked' => count($alerts)];
    }
    
    private function checkAndNotify($alert) {
        // Get current price
        $current_price = $this->getCurrentPrice($alert);
        
        // Update last checked
        $stmt = $this->db->prepare("
            UPDATE price_alerts 
            SET last_checked = NOW(), current_price = :price 
            WHERE id = :id
        ");
        
        $stmt->execute([
            ':price' => $current_price,
            ':id' => $alert['id']
        ]);
        
        // Check if price dropped
        if ($current_price < $alert['current_price']) {
            $drop_percentage = (($alert['current_price'] - $current_price) / $alert['current_price']) * 100;
            
            if ($drop_percentage > 5) { // At least 5% drop
                $this->sendPriceDropEmail($alert, $current_price);
            }
        }
    }

document.querySelectorAll('.counter button').forEach(btn => {
    btn.addEventListener('click', function() {
        const input = this.parentElement.querySelector('input');
        const current = parseInt(input.value);
        const min = parseInt(input.min);
        const max = parseInt(input.max);
        
        if (this.classList.contains('plus') && current < max) {
            input.value = current + 1;
        } else if (this.classList.contains('minus') && current > min) {
            input.value = current - 1;
        }
        
        updatePassengerCount();
    });
});

function updatePassengerCount() {
    const adults = parseInt(document.querySelector('input[name="adults"]').value);
    const children = parseInt(document.querySelector('input[name="children"]').value);
    const infants = parseInt(document.querySelector('input[name="infants"]').value);
    const total = adults + children + infants;
    
    document.getElementById('passenger-count').textContent = `${total} Yolcu`;
}

document.querySelectorAll('input[type="date"]').forEach(input => {
    const today = new Date().toISOString().split('T')[0];
    input.min = today;
    
    input.addEventListener('change', function() {
        // Update return date min based on departure
        if (this.name === 'departureDate') {
            const returnDate = document.querySelector('input[name="returnDate"]');
            if (returnDate) {
                returnDate.min = this.value;
            }
        }
        // Update checkout min based on checkin
        if (this.name === 'checkin') {
            const checkout = document.querySelector('input[name="checkout"]');
            if (checkout) {
                const nextDay = new Date(this.value);
                nextDay.setDate(nextDay.getDate() + 1);
                checkout.min = nextDay.toISOString().split('T')[0];
            }
        }
    });
});

let suggestionsTimeout;
document.querySelectorAll('input[name="origin"], input[name="destination"], input[name="pickupLocation"]').forEach(input => {
    input.addEventListener('input', function() {
        clearTimeout(suggestionsTimeout);
        const value = this.value;
        const suggestionBox = this.nextElementSibling;
        
        if (value.length < 2) {
            suggestionBox.classList.remove('show');
            return;
        }
        
        suggestionsTimeout = setTimeout(() => {
            // API call for suggestions
            fetch(`/api/autocomplete.php?q=${encodeURIComponent(value)}`)
                .then(response => response.json())
                .then(data => {
                    suggestionBox.innerHTML = '';
                    data.suggestions.forEach(item => {
                        const div = document.createElement('div');
                        div.className = 'suggestion-item';
                        div.innerHTML = `
                            <span class="icon">${item.type === 'airport' ? '✈️' : '🏙️'}</span>
                            <div class="details">
                                <div class="name">${item.name}</div>
                                <div class="code">${item.code}</div>
                            </div>
                        `;
                        div.addEventListener('click', () => {
                            input.value = `${item.name} (${item.code})`;
                            suggestionBox.classList.remove('show');
                        });
                        suggestionBox.appendChild(div);
                    });
                    suggestionBox.classList.add('show');
                });
        }, 300);
    });
});
