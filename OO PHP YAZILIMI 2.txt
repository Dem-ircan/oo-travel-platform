// assets/js/analytics.js
class OOAnalytics {
    constructor() {
        this.sessionId = this.generateSessionId();
        this.initTracking();
    }
    
    generateSessionId() {
        return 'oo_' + Date.now() + '_' + Math.random().toString(36).substr(2, 9);
    }
    
    initTracking() {
        // Track page views
        this.trackPageView();
        
        // Track search events
        this.trackSearches();
        
        // Track clicks
        this.trackClicks();
    }
    
    trackPageView() {
        gtag('event', 'page_view', {
            session_id: this.sessionId,
            page_title: document.title,
            page_location: window.location.href
        });
    }
    
    trackSearch(type, params) {
        // Send to our API
        fetch('/api/analytics.php', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                event: 'search',
                session_id: this.sessionId,
                search_type: type,
                params: params
            })
        });
        
        // Send to GA
        gtag('event', 'search', {
            search_term: `${params.origin}-${params.destination}`,
            search_type: type
        });
    }
    
    trackClick(element, data) {
        fetch('/api/analytics.php', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                event: 'click',
                session_id: this.sessionId,
                click_data: data
            })
        });
    }
    
    trackSearches() {
        document.querySelectorAll('.search-form').forEach(form => {
            form.addEventListener('submit', (e) => {
                const formData = new FormData(form);
                const type = form.getAttribute('data-type');
                const params = Object.fromEntries(formData);
                this.trackSearch(type, params);
            });
        });
    }
    
    trackClicks() {
        document.addEventListener('click', (e) => {
            if (e.target.matches('[data-track]')) {
                const data = e.target.dataset;
                this.trackClick(e.target, data);
            }
        });
    }
}

// Initialize
new OOAnalytics();

        if (!('webkitSpeechRecognition' in window)) return;
        
        const voiceBtn = document.getElementById('voiceSearchBtn');
        const recognition = new webkitSpeechRecognition();
        
        recognition.lang = 'tr-TR';
        recognition.onresult = (event) => {
            const result = event.results[0][0].transcript;
            this.processVoiceCommand(result);
        };
        
        voiceBtn?.addEventListener('click', () => {
            recognition.start();
        });
    }
    
    processVoiceCommand(command) {
        // Simple NLP for Turkish commands
        if (command.includes('u√ßak bileti')) {
            document.querySelector('[data-type="flight"]').click();
        } else if (command.includes('otel')) {
            document.querySelector('[data-type="hotel"]').click();
        }
        // More command processing...
    }
    
    setupPriceAlerts() {
        document.querySelectorAll('.price-alert-form').forEach(form => {
            form.addEventListener('submit', async (e) => {
                e.preventDefault();
                const formData = new FormData(form);
                
                try {
                    const response = await fetch('/api/price-alert.php', {
                        method: 'POST',
                        body: formData
                    });
                    
                    const result = await response.json();
                    
                    if (result.success) {
                        this.showNotification('Fiyat alarmƒ±nƒ±z kuruldu!', 'success');
                        form.reset();
                    }
                } catch (error) {
                    this.showNotification('Bir hata olu≈ütu.', 'error');
                }
            });
        });
    }
    
    setupLazyLoading() {
        if ('IntersectionObserver' in window) {

                entries.forEach(entry => {

                        img.src = img.dataset.src;
                        img.classList.remove('lazy');
                        observer.unobserve(img);
                    }
                });
            });

    '/',

    '/assets/js/app.min.js',
    '/assets/images/logo.webp',

        caches.open(CACHE_NAME)
            .then(cache => cache.addAll(urlsToCache))
    );
});

self.addEventListener('fetch', event => {
    event.respondWith(
        caches.match(event.request)
            .then(response => {
                // Return cached version or fetch from network
                return response || fetch(event.request);
            })
            .catch(() => {
                // Show offline page for navigation requests
                if (event.request.mode === 'navigate') {
                    return caches.match('/offline.html');
                }
            })
    );

// includes/security.php

class Security {
    
    public static function validateCSRF($token) {
        return isset($_SESSION['csrf_token']) && 
               hash_equals($_SESSION['csrf_token'], $token);
    }
    
    public static function generateCSRF() {
        $_SESSION['csrf_token'] = bin2hex(random_bytes(32));
        return $_SESSION['csrf_token'];
    }
    
    public static function sanitizeInput($data) {
        if (is_array($data)) {
            return array_map([self::class, 'sanitizeInput'], $data);
        }
        return htmlspecialchars(trim($data), ENT_QUOTES, 'UTF-8');
    }
    
    public static function rateLimitCheck($identifier, $limit = 100, $window = 3600) {
        $key = "rate_limit_{$identifier}";
        $current = get_cache($key) ?: 0;
        
        if ($current >= $limit) {
            http_response_code(429);
            die(json_encode(['error' => 'Rate limit exceeded']));
        }
        
        set_cache($key, $current + 1, $window);
        return true;
    }
    
    public static function validateEmail($email) {
        return filter_var($email, FILTER_VALIDATE_EMAIL) !== false;
    }
    
    public static function hashPassword($password) {
        return password_hash($password, PASSWORD_ARGON2ID);
    }
    
    public static function verifyPassword($password, $hash) {
        return password_verify($password, $hash);
    }
    
    public static function generateSecureToken($length = 32) {
        return bin2hex(random_bytes($length));
    }
}
