// assets/js/analytics.js
class OOAnalytics {
    constructor() {
        this.sessionId = this.generateSessionId();
        this.initTracking();
    }
    
    generateSessionId() {
        return 'oo_' + Date.now() + '_' + Math.random().toString(36).substr(2, 9);
    }
    
    initTracking() {
        // Track page views
        this.trackPageView();
        
        // Track search events
        this.trackSearches();
        
        // Track clicks
        this.trackClicks();
    }
    
    trackPageView() {
        gtag('event', 'page_view', {
            session_id: this.sessionId,
            page_title: document.title,
            page_location: window.location.href
        });
    }
    
    trackSearch(type, params) {
        // Send to our API
        fetch('/api/analytics.php', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                event: 'search',
                session_id: this.sessionId,
                search_type: type,
                params: params
            })
        });
        
        // Send to GA
        gtag('event', 'search', {
            search_term: `${params.origin}-${params.destination}`,
            search_type: type
        });
    }
    
    trackClick(element, data) {
        fetch('/api/analytics.php', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                event: 'click',
                session_id: this.sessionId,
                click_data: data
            })
        });
    }
    
    trackSearches() {
        document.querySelectorAll('.search-form').forEach(form => {
            form.addEventListener('submit', (e) => {
                const formData = new FormData(form);
                const type = form.getAttribute('data-type');
                const params = Object.fromEntries(formData);
                this.trackSearch(type, params);
            });
        });
    }
    
    trackClicks() {
        document.addEventListener('click', (e) => {
            if (e.target.matches('[data-track]')) {
                const data = e.target.dataset;
                this.trackClick(e.target, data);
            }
        });
    }
}

// Initialize
new OOAnalytics();

                // Return cached version or fetch from network

                    return caches.match('/offline.html');
                }
            })
    );

// includes/security.php

class Security {
    
    public static function validateCSRF($token) {
        return isset($_SESSION['csrf_token']) && 
               hash_equals($_SESSION['csrf_token'], $token);
    }
    
    public static function generateCSRF() {
        $_SESSION['csrf_token'] = bin2hex(random_bytes(32));
        return $_SESSION['csrf_token'];
    }
    
    public static function sanitizeInput($data) {
        if (is_array($data)) {
            return array_map([self::class, 'sanitizeInput'], $data);
        }
        return htmlspecialchars(trim($data), ENT_QUOTES, 'UTF-8');
    }
    
    public static function rateLimitCheck($identifier, $limit = 100, $window = 3600) {
        $key = "rate_limit_{$identifier}";
        $current = get_cache($key) ?: 0;
        
        if ($current >= $limit) {
            http_response_code(429);
            die(json_encode(['error' => 'Rate limit exceeded']));
        }
        
        set_cache($key, $current + 1, $window);
        return true;
    }
    
    public static function validateEmail($email) {
        return filter_var($email, FILTER_VALIDATE_EMAIL) !== false;
    }
    
    public static function hashPassword($password) {
        return password_hash($password, PASSWORD_ARGON2ID);
    }
    
    public static function verifyPassword($password, $hash) {
        return password_verify($password, $hash);
    }
    
    public static function generateSecureToken($length = 32) {
        return bin2hex(random_bytes($length));
    }
}
